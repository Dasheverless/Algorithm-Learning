
### 思路
1.认识到题解是 $C_3^x + C_6^x + ... + C_{3n}^x$  
2.联想到帕斯卡恒等式 $C_n^k = C_{n-1}^{k-1} + C_{n-1}^k$  
3.将其扩展到$C_0^x + C_1^x + C_2^x +  C_3^x + ... + C_{3n-1}^x + C_{3n}^x$  
4.归纳为$C_{3n}^{x+1}$  
5.利用帕斯卡不等式做dp
```C++
#include <bits/stdc++.h>
#include <algorithm>
using namespace std;
typedef long long ll;
using vi = vector<int>;
using vl = vector<ll>;
using vvl = vector< vector<ll> >;
using vvi = vector< vector<int> >;
const ll mod = 1e9 + 7;

ll quickpow(ll a, ll b)
{
    a %= mod;
    ll ans = 1;// ans：幂的结果；base：底数a
    while (b)
    {
        if (b & 1)
        {
            ans = (ans * a)%mod;
        }
        a = (a * a)%mod;
        b = b >> 1;
    }
    return ans;
}
vl fac, ifac;
void init(int n)
{
    fac.resize(n + 1);
    ifac.resize(n + 1);
    fac[0] = 1, fac[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        fac[i] = (fac[i - 1] * i) % mod;
    }
    ifac[n] = quickpow(fac[n], mod - 2);
    for (int i = n - 1; i >= 0; i--)
    {
        ifac[i] = (ifac[i + 1] * (i + 1)) % mod;
    }
}
ll C(int n, int r)
{
    if (n < 0 || r < 0 || n < r) return 0LL;
    ll inv = (ifac[r] * ifac[n - r]) % mod;
    return (fac[n] * inv)%mod;
}
int main()
{
    /*
    思路：1认识到题解是C_3^x + C_6^x + ... + C_3n^x
    2.
    */
    ios::sync_with_stdio(false);
    cin.tie(0);
    int n, q, x;
    cin >> n;
    init(3 * n + 1);
    ll div3 = quickpow(3, mod - 2);
    vvl dp(3 * n + 1, vl(3));
    dp[0][0] = dp[0][1] = dp[0][2] = n;
    for (int i = 1; i < 3 * n; i++)
    {
        dp[i][0] = ((C(3*n,i+1) - 2 * dp[i - 1][0] - dp[i - 1][1] + 3*mod) * div3) % mod;
        dp[i][1] = (dp[i][0] + dp[i - 1][0]) % mod;
        dp[i][2] = (dp[i][1] + dp[i - 1][1]) % mod;
    }
    cin >> q;
    while (q--)
    {
        cin >> x;
        if (x == 3 * n)
            printf("1\n");
        else
            printf("%lld\n", (dp[x][0] + C(3 * n, x))%mod);
    }
}
```
